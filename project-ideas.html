  • 4. Password Generator

    A script that generates strong passwords with customizable length and character types.

    # Password Generator (Requires random module)

import random
import string

def generate_password(length=12):
    characters = string.ascii_letters + string.digits + string.punctuation
    password = ''.join(random.choice(characters) for i in range(length))
    return password

length = int(input("Enter password length: "))

  • 7. File Organizer

    A script to automatically organize files in a directory into subdirectories based on file types or extensions.

    # File Organizer (Requires os module)

import os
import shutil

def organize_files(directory):
    for filename in os.listdir(directory):
        file_extension = filename.split('.')[-1]
        target_directory = os.path.join(directory, file_extension)

        if not os.path.exists(target_directory):
            os.makedirs(target_directory)

        source = os.path.join(directory, filename)
        destination = os.path.join(target_directory, filename)

        shutil.move(source, destination)

directory = input("Enter directory path to organize: ")
organize_files(directory)

  • 9. Currency Converter

    A tool to convert between different currencies using current exchange rates.

    # Currency Converter (Requires requests module)

import requests

def convert_currency(amount, from_currency, to_currency):
    api_key = 'your_api_key_here'
    base_url = 'https://api.exchangerate-api.com/v4/latest/'
    complete_url = f"{base_url}{from_currency}"

    response = requests.get(complete_url)
    data = response.json()
    rates = data['rates']

    if to_currency in rates:
        converted_amount = amount * rates[to_currency]
        return converted_amount
    else:
        return "Currency not found."

amount = float(input("Enter amount: "))
from_currency = input("From currency (e.g., USD): ").upper()
to_currency = input("To currency (e.g., EUR): ").upper()
converted_amount = convert_currency(amount, from_currency, to_currency)
print(f"Converted amount: {converted_amount}")

  • 10. Sudoku Solver

    A program that solves Sudoku puzzles using algorithms.

    # Sudoku Solver

def print_board(board):
    for row in board:
        print(" ".join(str(num) if num != 0 else '.' for num in row))

def is_valid(board, row, col, num):
    for i in range(9):
        if board[row][i] == num or board[i][col] == num:
            return False
    start_row, start_col = 3 * (row // 3), 3 * (col // 3)
    for i in range(start_row, start_row + 3):
        for j in range(start_col, start_col + 3):
            if board[i][j] == num:
                return False
    return True

def solve_sudoku(board):
    for row in range(9):
        for col in range(9):
            if board[row][col] == 0:
                for num in range(1, 10):
                    if is_valid(board, row, col, num):
                        board[row][col] = num
                        if solve_sudoku(board):
                            return True
                        board[row][col] = 0
                return False
    return True

# Example Sudoku board (0 represents empty cells)
board = [
    [5, 3, 0, 0, 7, 0, 0, 0, 0],
    [6, 0, 0, 1, 9, 5, 0, 0, 0],
    [0, 9, 8, 0, 0, 0, 0, 6, 0],
    [8, 0, 0, 0, 6, 0, 0, 0, 3],
    [4, 0, 0, 8, 0, 3, 0, 0, 1],
    [7, 0, 0, 0, 2, 0, 0, 0, 6],
    [0, 6, 0, 0, 0, 0, 2, 8, 0],
    [0, 0, 0, 4, 1, 9, 0, 0, 5],
    [0, 0, 0, 0, 8, 0, 0, 7, 9]
]

  • 11. Tic-Tac-Toe Game

    A simple implementation of the classic Tic-Tac-Toe game for two players.

    # Tic-Tac-Toe Game

def print_board(board):
    for row in board:
        print(" | ".join(row))
        print("-" * 5)

def check_winner(board, mark):
    for row in board:
        if all(cell == mark for cell in row):
            return True
    for col in range(3):
        if all(board[row][col] == mark for row in range(3)):
            return True
    if all(board[i][i] == mark for i in range(3)) or all(board[i][2-i] == mark for i in range(3)):
        return True
    return False

def tic_tac_toe():
    board = [[' ']*3 for _ in range(3)]
    current_player = 'X'
    for _ in range(9):
        print_board(board)
        row = int(input(f"Player {current_player}, enter row (0-2): "))
        col = int(input(f"Player {current_player}, enter column (0-2): "))
        if board[row][col] == ' ':
            board[row][col] = current_player
            if check_winner(board, current_player):
                print_board(board)
                print(f"Player {current_player} wins!")
                return
            current_player = 'O' if current_player == 'X' else 'X'
        else:
            print("Cell already taken, try again.")
    print_board(board)
    print("It's a draw!")

tic_tac_toe()

  • 13. Library Management System

    An application to manage book checkouts and returns in a library.

    # Library Management System

library = {
    'The Great Gatsby': True,
    '1984': True,
    'To Kill a Mockingbird': True
}

def checkout_book(book_title):
    if library.get(book_title):
        library[book_title] = False
        print(f"You have checked out '{book_title}'.")
    else:
        print(f"'{book_title}' is not available or does not exist.")

def return_book(book_title):
    if book_title in library:
        library[book_title] = True
        print(f"'{book_title}' has been returned.")
    else:
        print(f"'{book_title}' does not exist in the library.")

def show_books():
    for book, available in library.items():
        status = 'Available' if available else 'Checked Out'
        print(f"{book}: {status}")

while True:
    print("\n1. Checkout Book")
    print("2. Return Book")
    print("3. Show Books")
    print("4. Exit")
    choice = input("Enter choice: ")

    if choice == '1':
        title = input("Enter book title to checkout: ")
        checkout_book(title)
    elif choice == '2':
        title = input("Enter book title to return: ")
        return_book(title)
    elif choice == '3':
        show_books()
    elif choice == '4':
        break
    else:
        print("Invalid choice. Please enter a number between 1 and 4.")

  • 14. Alarm Clock

    A program that allows users to set alarms and notifies them at the specified time.

    # Alarm Clock

import time
import threading

def set_alarm(alarm_time):
    while True:
        current_time = time.strftime('%H:%M:%S')
        if current_time == alarm_time:
            print("Alarm ringing!")
            break
        time.sleep(1)

alarm_time = input("Enter alarm time (HH:MM:SS): ")
alarm_thread = threading.Thread(target=set_alarm, args=(alarm_time,))
alarm_thread.start()
